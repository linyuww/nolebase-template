二叉搜索树中的众数

出现频率最高

先遍历排序出来，如何找最多的

只遍历一次使用**双指针**的方法

```cpp
TreeNode* pre = nullptr;
    int count = 0;
    int maxcount = 0;
    vector<int> res;
    void travel(TreeNode* root){

        if(root == nullptr)return;

        travel(root -> left);
        if(pre == nullptr)//第一次到最左叶子结点
            count=1;
        else if(pre -> val == root -> val)//前后相同说明频数就加一
            count++;
        else count = 1;//不同就归一
        
        pre = root;//
        
        if(count == maxcount){
            res.push_back(root -> val);
        }//不管怎么样，先存进去
        if(count > maxcount)
        //超过了，说明原来存的所有的都不符合条件
        //如果是同一个数一直出现，就一直更新这里直到前面count=1就不执行这里，对应下面root和pre指向同一个，不会不一样。
        {
            maxcount = count;
            res.clear();
            res.push_back(root -> val);//这里root和pre指向的是同一个结点
        }

        travel(root -> right);

    }
    vector<int> findMode(TreeNode* root) {

        travel(root);
        return res;
    }
```

```cpp
vector<int> res;
    void travel(TreeNode* root)
    {
        if(root == nullptr)return ;

        travel(root -> left);
        res.push_back(root -> val);
        travel(root -> right);

    }
    vector<int> findMode(TreeNode* root) {

        travel(root);
        int ans = 0;
        unordered_map<int,int> mp;
        for(int i = 0;i < res.size() ;i++){
            mp[res[i]]++;
        }

        for(int i = 0;i < res.size();i++){
            ans = max (mp[res[i]], ans);
        }
        vector<int> r;
        for(int i = 0;i < res.size();i++){
            
            if(r.empty() && mp[res[i]] == ans)r.push_back(res[i]);
            else if(mp[res[i]] == ans && r.back() != res[i])r.push_back(res[i]);
        }

        return r;

    }
```
